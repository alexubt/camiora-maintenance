<feature>
  <name>Schedule Calculation Functions</name>
  <files>app/maintenance/schedule.js, app/maintenance/schedule.test.js</files>
  <behavior>
    getDueDate:
    - getDueDate({ LastDoneDate: '2025-09-01', IntervalDays: '365' }) -> '2026-09-01'
    - getDueDate({ LastDoneDate: '2026-01-31', IntervalDays: '30' }) -> '2026-03-02'
    - getDueDate({ LastDoneDate: '', IntervalDays: '90' }) -> null
    - getDueDate({ LastDoneDate: '2025-09-01', IntervalDays: '' }) -> null

    getDueMiles:
    - getDueMiles({ LastDoneMiles: '145200', IntervalMiles: '10000' }) -> 155200
    - getDueMiles({ LastDoneMiles: '', IntervalMiles: '10000' }) -> null
    - getDueMiles({ LastDoneMiles: '145200', IntervalMiles: '' }) -> null

    isOverdue (time-based):
    - isOverdue({ LastDoneDate: '2025-01-01', IntervalDays: '90' }, '2026-03-17', null) -> true (way past due)
    - isOverdue({ LastDoneDate: '2026-03-01', IntervalDays: '90' }, '2026-03-17', null) -> false (due 2026-05-30)
    - isOverdue({ LastDoneDate: '2026-03-01', IntervalDays: '15' }, '2026-03-17', null) -> true (due 2026-03-16, today is 17th)
    - isOverdue({ LastDoneDate: '2026-03-01', IntervalDays: '16' }, '2026-03-17', null) -> false (due 2026-03-17, today equals due date)

    isOverdue (mileage-based):
    - isOverdue({ LastDoneMiles: '100000', IntervalMiles: '10000' }, '2026-03-17', 110000) -> true (at limit)
    - isOverdue({ LastDoneMiles: '100000', IntervalMiles: '10000' }, '2026-03-17', 109999) -> false (one mile short)

    isOverdue (missing data):
    - isOverdue({ IntervalDays: '', LastDoneDate: '' }, '2026-03-17', null) -> false (no schedule configured)
    - isOverdue({ IntervalDays: '90' }, '2026-03-17', null) -> false (no LastDoneDate)
  </behavior>
  <implementation>
    Create `app/maintenance/schedule.js` as a native ES module exporting three pure functions:

    1. `getDueDate(record)` — Computes due date string from `record.LastDoneDate` + `record.IntervalDays`. Returns YYYY-MM-DD string or null. Uses `new Date(dateStr + 'T00:00:00')` to avoid timezone offset issues (Pitfall 1 from research). Uses `setDate(getDate() + N)` for day arithmetic.

    2. `getDueMiles(record)` — Computes due mileage from `record.LastDoneMiles` + `record.IntervalMiles`. Returns number or null.

    3. `isOverdue(record, todayStr, currentMiles)` — Returns boolean. Checks time-based overdue (today > dueDate using string comparison since both are YYYY-MM-DD) and mileage-based overdue (currentMiles >= dueMiles). Returns false if no interval/last-done data. `currentMiles` can be null/undefined for units without mileage tracking.

    All values from CSV are strings — functions must `Number()` convert as needed. Follow the exact signatures from 04-RESEARCH.md Pattern 1.
  </implementation>
</feature>
